3. Como Recriar o Dashboard do Zero

1
Criar conta no GitHub (se não tiver) em github.com — conta gratuita é suficiente.
2
Criar repositório público: clicar em "+" → "New repository" → dar um nome (ex: meu-dashboard) → marcar "Public" → "Create repository".
3
Ativar o GitHub Pages: Settings → Pages → Source: Deploy from a branch → Branch: main → / (root) → Save. O link será: https://SEU-USUARIO.github.io/NOME-DO-REPO/
4
Criar o index.html: clicar em "Add file" → "Create new file" → nomear index.html → colar o código-fonte completo da Seção 6 deste documento → "Commit changes".
5
Aguardar deploy (30 a 90 segundos). Acompanhar na aba "Actions" do repositório. Ícone verde ✓ = publicado e acessível.
6
Configurar o dashboard: abrir o link → preencher token Airtable, IDs de base/tabela/view e nomes exatos dos campos de data → clicar "Carregar Dados".

5. Pontos Críticos da Lógica

5.1 API retorna campos por NOME, não por ID

// CORRETO — usar o nome exato do campo como chave:
const start = record.fields["Data Avaliação"];
const end   = record.fields["Data Alta Cardiológica"];
// ERRADO — a API NÃO usa IDs de campo (fldXXX...)

5.2 Suavização APENAS no Período Total + Diário

// Anos individuais usam dados brutos (preserva dias de baixa demanda)
if (CUR_YEAR === "total" && CUR_VIEW === "diario") {
  vals  = rollingAvg(keys, dm, 14);
  label = "Média móvel 14 dias";
} else {
  vals  = keys.map(k => dm[k] || 0);
  label = "Casos ativos por dia";
}

5.3 Registro sem campo de fim = caso ainda ativo (usa data atual)

const ed = record.fields[endField]
  ? new Date(record.fields[endField])
  : new Date(); // usa a data de hoje se o campo estiver vazio

5.4 Evitar SyntaxError — data-attributes em vez de onclick inline

// CORRETO:
tab.dataset.year = year;
tab.addEventListener("click", e => setYear(e.currentTarget.dataset.year));
// ERRADO: gera SyntaxError quando há aspas simples dentro de atributo double-quoted

function $(id){ return document.getElementById(id); }
function showErr(id,msg){ var el=$(id); if(!el)return; el.textContent=msg; el.style.display=msg?"block":"none"; }

function startLoad(){
  var t=$("fToken").value.trim(), b=$("fBase").value.trim();
  var tb=$("fTable").value.trim(), v=$("fView").value.trim();
  var fs=$("fStart").value.trim(), fe=$("fEnd").value.trim();
  if(!t||!b||!tb||!fs||!fe){ showErr("cfgErr","Preencha todos os campos obrigatórios."); return; }
  showErr("cfgErr","");
  $("btnLoad").textContent="Carregando…"; $("btnLoad").disabled=true;
  CFG.token=t; CFG.base=b; CFG.table=tb; CFG.view=v; CFG.fieldStart=fs; CFG.fieldEnd=fe;
  LS.save("dash_token",t); LS.save("dash_base",b); LS.save("dash_table",tb);
  LS.save("dash_view",v); LS.save("dash_start",fs); LS.save("dash_end",fe);
  fetchPage([],null);
}

function fetchPage(acc,offset){
  var url="https://api.airtable.com/v0/"+CFG.base+"/"+CFG.table+"?pageSize=100";
  if(CFG.view) url+="&view="+encodeURIComponent(CFG.view);
  if(offset) url+="&offset="+encodeURIComponent(offset);
  fetch(url,{headers:{Authorization:"Bearer "+CFG.token}})
    .then(function(r){return r.json();})
    .then(function(d){
      if(d.error){ showErr("cfgErr","Erro da API: "+d.error.message); $("btnLoad").textContent="Carregar Dados"; $("btnLoad").disabled=false; return; }
      var recs=d.records||[];
      for(var i=0;i<recs.length;i++) acc.push(recs[i]);
      if(d.offset) fetchPage(acc,d.offset);
      else buildDashboard(acc);
    })
    .catch(function(e){ showErr("cfgErr","Erro de rede: "+e.message); $("btnLoad").textContent="Carregar Dados"; $("btnLoad").disabled=false; });
}

function buildDashboard(recs){
  var today=new Date(); today.setHours(0,0,0,0);
  var daily={}, valid=0;
  for(var i=0;i<recs.length;i++){
    var f=recs[i].fields, sv=f[CFG.fieldStart];
    if(!sv) continue;
    var sd=new Date(sv); sd.setHours(0,0,0,0);
    if(isNaN(sd.getTime())) continue;
    var ev=f[CFG.fieldEnd], ed=ev?new Date(ev):new Date(today);
    ed.setHours(0,0,0,0);
    if(isNaN(ed.getTime())) ed=new Date(today);
    if(ed<sd) ed=new Date(sd);
    valid++;
    var cur=new Date(sd);
    while(cur<=ed){ var k=cur.toISOString().slice(0,10); daily[k]=(daily[k]||0)+1; cur.setDate(cur.getDate()+1); }
  }
  if(!valid){ showErr("cfgErr","Nenhum registro válido. Verifique os nomes dos campos."); $("btnLoad").textContent="Carregar Dados"; $("btnLoad").disabled=false; return; }
  DAILY=daily;
  var ySet={}; Object.keys(daily).forEach(function(k){ySet[k.slice(0,4)]=1;});
  YEARS=Object.keys(ySet).sort();
  CUR_YEAR="total"; CUR_VIEW="diario";
  $("appSub").textContent=valid+" registros · "+Object.keys(daily).length+" dias";
  $("pg-cfg").style.display="none"; $("pg-app").style.display="block";
  $("btnLoad").textContent="Carregar Dados"; $("btnLoad").disabled=false;
  renderAll();
}

function getFiltered(){
  if(CUR_YEAR==="total") return DAILY;
  var r={}; Object.keys(DAILY).forEach(function(k){if(k.slice(0,4)===CUR_YEAR)r[k]=DAILY[k];}); return r;

function rollingAvg(keys,dm,window){
  var half=Math.floor(window/2);
  return keys.map(function(k,i){
    var sum=0, count=0;
    for(var j=Math.max(0,i-half); j<=Math.min(keys.length-1,i+half); j++){ sum+=dm[keys[j]]; count++; }
    return Math.round(sum/count*10)/10;
  });
}

function buildChartData(dm){
  var keys=Object.keys(dm).sort();
  if(!keys.length) return {labels:[],data:[],legend:""};

  if(CUR_VIEW==="diario"){
    var raw=keys.map(function(k){return dm[k];});
    /* Smooth ONLY for Periodo Total (too many points to read clearly) */
    if(CUR_YEAR==="total"){
      var smoothed=rollingAvg(keys,dm,14);
      return {labels:keys, data:smoothed, legend:"Média móvel 14 dias"};
    }
    /* Individual years: show raw daily data */
    return {labels:keys, data:raw, legend:"Casos ativos por dia"};
  }

  if(CUR_VIEW==="semanal"){
    var wk={};
    keys.forEach(function(k){
      var d=new Date(k),day=d.getDay(),diff=d.getDate()-day+(day===0?-6:1),m=new Date(d);
      m.setDate(diff); m.setHours(0,0,0,0);
      var wkey=m.toISOString().slice(0,10);
      if(!wk[wkey])wk[wkey]=[];
      wk[wkey].push(dm[k]);
    });
    var wkeys=Object.keys(wk).sort();
    return {labels:wkeys, data:wkeys.map(function(k){return Math.max.apply(null,wk[k]);}), legend:"Máximo semanal"};
  }

  var mo={};
  keys.forEach(function(k){var mk=k.slice(0,7);if(!mo[mk])mo[mk]=[];mo[mk].push(dm[k]);});
  var mkeys=Object.keys(mo).sort();
  if(CUR_VIEW==="mensal-max") return {labels:mkeys, data:mkeys.map(function(k){return Math.max.apply(null,mo[k]);}), legend:"Máximo mensal"};
  if(CUR_VIEW==="mensal-avg") return {labels:mkeys, data:mkeys.map(function(k){
    var a=mo[k]; return Math.round(a.reduce(function(x,y){return x+y;},0)/a.length*10)/10;
  }), legend:"Média mensal"};
  if(CUR_VIEW==="mensal-mdn") return {labels:mkeys, data:mkeys.map(function(k){
    var a=mo[k].slice().sort(function(x,y){return x-y;}),mid=Math.floor(a.length/2);
    return a.length%2?a[mid]:Math.round((a[mid-1]+a[mid])/2*10)/10;
  }), legend:"Mediana mensal"};
  return {labels:[],data:[],legend:""};
}

function mdn(arr){var a=arr.slice().sort(function(x,y){return x-y;}),mid=Math.floor(a.length/2);return a.length%2?a[mid]:Math.round((a[mid-1]+a[mid])/2*10)/10;}

function renderChart(){
  var dm=getFiltered();
  var vals=Object.keys(dm).map(function(k){return dm[k];});
  if(!vals.length){ $("statsGrid").innerHTML='<p style="color:#94a3b8;font-size:13px">Sem dados para este período.</p>'; return; }
  var today=new Date().toISOString().slice(0,10);
  var mx=Math.max.apply(null,vals);
  var avg=Math.round(vals.reduce(function(a,b){return a+b;},0)/vals.length*10)/10;
  var med=mdn(vals);
  var active=dm[today]||0;
  var statsData=[
    {val:mx,    lbl:"Pico Máximo",      unit:"casos / dia",    c:STAT_COLORS[0]},
    {val:avg,   lbl:"Média Diária",    unit:"casos / dia",    c:STAT_COLORS[1]},
    {val:med,   lbl:"Mediana Diária",  unit:"casos / dia",    c:STAT_COLORS[2]},
    {val:active,lbl:"Casos Ativos Hoje",unit:"em andamento",   c:STAT_COLORS[3]},
    {val:vals.length,lbl:"Dias Analisados",unit:"dias no período",c:STAT_COLORS[4]}
  ];
  var html="";
  for(var i=0;i<statsData.length;i++){
    var s=statsData[i];
    html+='<div class="stat-card"><div class="stat-card-inner">';
    html+='<div class="stat-dot" style="background:'+s.c.dot+'"></div>';
    html+='<div class="stat-val" style="color:'+s.c.val+'">'+s.val+'</div>';
    html+='<div class="stat-lbl">'+s.lbl+'</div><div class="stat-unit">'+s.unit+'</div>';
    html+='</div></div>';
  }
  $("statsGrid").innerHTML=html;

  var cd=buildChartData(dm);
  $("chartTitle").textContent=cd.legend;
  var ctx=$("myChart").getContext("2d");
  if(CHART){CHART.destroy();CHART=null;}

  CHART=new Chart(ctx,{
    type:"bar",
    data:{labels:cd.labels,datasets:[{
      label:cd.legend,
      data:cd.data,
      backgroundColor:"rgba(79,70,229,0.72)",
      borderColor:"rgba(79,70,229,0.90)",
      borderWidth:1,
      borderRadius:2,
      borderSkipped:false
    }]},
    options:{
      responsive:true,
      plugins:{legend:{display:false},tooltip:{callbacks:{label:function(ctx){return " "+ctx.parsed.y+" casos";}}}},
      scales:{
        x:{ticks:{color:"#94a3b8",font:{size:11},maxTicksLimit:18},grid:{color:"#f1f5f9"},border:{color:"#e2e8f0"}},
        y:{ticks:{color:"#94a3b8",font:{size:11}},grid:{color:"#f1f5f9"},border:{color:"#e2e8f0"},beginAtZero:true}
      }
    }
  });
}
